import { useState } from 'react' import { validateForm } from '@/utils/validation' import { authApi } from '@/api/auth' export default function LoginPage() { const [email, setEmail] = useState('') const [password, setPassword] = useState('') const handleSubmit = async () => { if (!validateForm({ email, password })) return await authApi.login({ email, password }) } return ( <div className="flex min-h-screen items-center justify-center"> <form onSubmit={handleSubmit}> <Input value={email} onChange={setEmail} placeholder="Email" /> <Input type="password" value={password} onChange={setPassword} /> <Button type="submit">Sign In</Button> </form> </div> ) }
